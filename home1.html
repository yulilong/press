 * 本地模拟数据

// 只有本地开发才引入Mock
// const Mock = process.env.NODE_ENV === 'development' && require('mockjs')
import auth from './getAuth.ajax.js';

if (Mock) {
  // 拦截获取用户信息API
  Mock.mock('/ac/getAuth.ajax','post', auth);
}
